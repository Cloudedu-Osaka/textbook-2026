1.13.1 作図#

using SymPyPythonCall

@syms x, a, p, q

@syms f(), g(), h()

@syms g_1(), g_2(), g_3()

@ggb a=Slider(-2π, 2π)

@ggb g_1 = sin(x+a)

@ggb g_2 = sin(x+2a)

@ggb g_3 = sin(x+3a)

@ggb g = g_1(x) + g_2(x) + g_3(x)

@ggb g_1(x)+g_2(x)

@ggb g_2(x)+g_3(x)

@ggb g_3(x)+g_1(x)
